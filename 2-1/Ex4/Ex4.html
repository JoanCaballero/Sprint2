Rest & Spread Operators

Exercici 4.1

Declaració de la funció:

        
          const array1 = [1, 3, 5, 7, 9]
          const array2 = [2, 4, 6, 8, 10]
          const array = [...array1, ...array2]

Codi d'execució:

        
          console.log(array)

Exercici 4.2

Declaració de la funció:

        
          const suma = (...array) => {
            let total = 0
            for (let num of array) {
              total += num
            }
            return total
          }

Codi d'execució:

        
          const nums = [1, 2, 3, 4, 5]
          console.log(suma(...nums))

Exercici 4.3

Declaració de la funció:

        
          const quadrat = {
            costat: '2',
            colorMarge: 'vermell',
            colorInterior: 'blau',
          }
          const quadrat2 = { ...quadrat }
          quadrat2.costat = '3'

Codi d'execució:

        
          console.log(quadrat)
          console.log(quadrat2)

Exercici 4.4

Declaració de la funció:

        
          const noms = ['Joan', 'Judit', 'Xavi', 'Manel', 'Pumba']
          const [nom1, nom2, ...altresNoms] = noms

Codi d'execució:

        
          console.log(`Primer nom: ${nom1}`)
          console.log(`Segon nom: ${nom2}`)
          console.log(`Resta de noms: ${altresNoms}`)

Exercici 4.5

Declaració de la funció:

        
          const multiplicar3Nums = (a, b, c) => {
            return a * b * c
          }

Codi d'execució:

        
          const tresNums = [2, 3, 4]
          console.log(multiplicar3Nums(...tresNums))
